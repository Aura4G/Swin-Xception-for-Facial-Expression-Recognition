 42def extract_features(model, dataloader, device):
 43    """
 44    Freezes the passed model's and completes single pass of a dataset's images through the feature-extractor backbone, and compiles
 45    the resulting features (along with their corresponding labels) into data structures.
 46
 47    Args:
 48        model (nn.Module): The Swin-Xception model that performs the feature extraction.
 49        dataloader (DataLoader): The batch-processed dataset used for the feature extraction.
 50        device (torch.device): Defines which device the model's feature extraction is being ran on.
 51
 52    Returns:
 53        features (List): Data structure containing all of the extracted features for each class.
 54        labels (List): The data structure of labels that parallel the features data structure.
 55    """
 56
 57    model.eval()
 58    all_features = []
 59    all_labels = []
 60
 61    with torch.no_grad():
 62        for images, labels in tqdm(dataloader, desc="Extracting Features"):
 63            images = images.to(device)
 64
 65            x = model.patch_embed(images)
 66
 67            for block in model.layer1:
 68                x = block(x)
 69            x = model.merge1(x)
 70
 71            for block in model.layer2:
 72                x = block(x)
 73            x = model.merge2(x)
 74
 75            for block in model.layer3:
 76                x = block(x)
 77            x = model.merge3(x)
 78            
 79            for block in model.layer4:
 80                x = block(x)
 81
 82            x = model.norm(x)
 83            x = x.transpose(1, 2)
 84            x = model.avgpool1d(x)
 85            x = torch.flatten(x, 1)
 86
 87            all_features.append(x.cpu().numpy())
 88            all_labels.append(labels.numpy())
 89
 90        features = np.vstack(all_features)
 91        labels = np.concatenate(all_labels)
 92
 93        return features, labels

 95def apply_smote(features, labels, random_state=42):
 96    """
 97    Counts total feature support per-class and applies SMOTE to all classes except the majority to create uniform support.
 98
 99    Args:
100        features (List): A list of features extracted from the frozen Swin-Xception backbone.
101        labels (List): A list of labels corresponding to the features list
102        random_state (int): Sets the random seed for the SMOTE algorithm. Default: 42, to match all other random seed instantiations.
103
104    Returns:
105        balanced_features (List): A list of features, having been supplied synthetic samples of minority classes
106        balanced_labels (List): A list of labels, parallel to the feature list.
107    """
108
109    # Count every feature extracted by class
110    print("Original class distribution:")
111    print(Counter(labels))
112    
113    smote = SMOTE(random_state=random_state, k_neighbors=5) # Uses 5 nearest neighbours to interpolate new samples
114    balanced_features, balanced_labels = smote.fit_resample(features, labels) # smote.fit_resample returns the lists of balanced features and labels.
115
116    print("Balanced class distribution:")
117    print(Counter(balanced_labels))
118
119    return balanced_features, balanced_labels

124def crop_to_square(img):
125    """
126    Crops non-standardised images (images that are not natively from a dataset) to a square
127    
128    Args:
129        img: Variably-sized image input
130
131    Returns:
132        img: Image, where height = width
133    """
134
135    h, w = img.shape[:2]
136    
137    if w > h:
138        start_x = (w - h) // 2
139        return img[:, start_x : start_x + h]
140    elif h > w:
141        start_y = (h - w) // 2
142        return img[start_y : start_y + w, :]
143    else:
144        return img

146def get_face_crop(image):
147    """ Takes a non-standardised image and uses a Haar cascade classifier to locate a face and crop the image to the face. """
148
149    grey = cv2.cvtColor(image, cv2.COLOR_BGR2GRAY)
150    faces = face_cascade.detectMultiScale(grey, 1.3, 5)
151
152    if len(faces) > 0:
153        x, y, w, h = max(faces, key=lambda b: b[2] * b[3])
154
155        side = max(w, h)
156        center_x, center_y = x + w//2, y + h//2
157
158        nx1 = max(0, center_x - side//2)
159        ny1 = max(0, center_y - side//2)
160        nx2 = min(image.shape[1], nx1 + side)
161        ny2 = min(image.shape[0], ny1 + side)
162
163        return image[ny1:ny2, nx1:nx2]
164
165    return image

167def save_activations(module, input, output):
168    """
169    Hook function to capture the output activations of a registered module.
170    Stores the output tensor in the external `data` dict under 'activations'.
171
172    Args:
173        module (nn.Module): The module the hook is registered to.
174        input (tuple): Inputs to the module (unused).
175        output (torch.Tensor): Output activations of the module.
176    """
177    data["activations"] = output

179def save_gradients(module, grad_input, grad_output):
180    """
181    Hook function to capture the gradient of the loss with respect to a 
182    registered module's output. Stores the gradient in the external `data` 
183    dict under 'gradients'.
184
185    Args:
186        module (nn.Module): The module the hook is registered to.
187        grad_input (tuple): Gradients with respect to the module's inputs (unused).
188        grad_output (tuple[torch.Tensor]): Gradients with respect to the 
189                                           module's output; first element is stored.
190    """
191    data["gradients"] = grad_output[0]

193def compute_heatmap(model, img_tensor):
194    """
195    Computes a Grad-CAM heatmap for the most confident predicted class.
196    Performs a forward and backward pass to collect activations and gradients
197    via registered hooks, then computes a weighted activation map.
198
199    Args:
200        model (nn.Module): The model to explain. Must have save_activations and 
201                           save_gradients hooks registered prior to calling.
202        img_tensor (torch.Tensor): Input image tensor of shape (1, C, H, W).
203
204    Returns:
205        tuple:
206            - cam (np.ndarray): Heatmap of shape (H', W') where H' and W' are 
207                                the spatial dimensions of the activation layer.
208            - prediction (int): Index of the predicted class.
209    """
210
211    model.zero_grad()
212
213    output = model(img_tensor)
214    _, prediction = output.max(1)
215    
216
217    output[0, prediction].backward()
218    
219
220    acts = data["activations"]
221    grads = data["gradients"]
222    
223
224    weights = torch.mean(grads, dim=1, keepdim=True)
225
226    cam = torch.sum(weights * acts, dim=-1)
227
228    b, l = cam.shape
229    side = int(l**0.5) 
230    cam = cam.view(b, side, side)
231    
232    cam = torch.relu(cam)
233    cam = cam.detach().cpu().squeeze().numpy()
234    
235    return cam, prediction.item()

237def upsample_heatmap(heatmap, image):
238    """
239    Upsamples and overlays a Grad-CAM heatmap onto the original image.
240    Handles both torch.Tensor and np.ndarray inputs for both arguments,
241    normalises the heatmap to [0, 255], resizes it to match the image 
242    resolution, and blends them using a 70/30 heatmap/image weighting.
243
244    Args:
245        heatmap (torch.Tensor | np.ndarray): Raw CAM output of shape (H', W'), 
246                                             where H' and W' are the spatial 
247                                             dimensions of the activation layer.
248        image (torch.Tensor | np.ndarray): Original input image. If a Tensor, 
249                                           expected shape is (1, C, H, W) or 
250                                           (C, H, W) in RGB format.
251
252    Returns:
253        np.ndarray: RGB image of shape (H, W, 3) with the heatmap overlaid 
254                    on the original image.
255    """
256
257    if isinstance(image, torch.Tensor):
258        image = image.detach().cpu().squeeze()
259        if image.dim() == 3:
260            image = image.permute(1, 2, 0).numpy()
261        else:
262            image = image.numpy()
263
264    if torch.is_tensor(heatmap):
265        heatmap = torch.maximum(heatmap, torch.tensor(0.0))
266        heatmap = heatmap.numpy()
267    else:
268        heatmap = np.maximum(heatmap, 0)
269
270    m, M = heatmap.min(), heatmap.max()
271    if M - m > 0:
272        heatmap_norm = 255 * ((heatmap - m) / (M - m))
273    else:
274        heatmap_norm = np.zeros_like(heatmap)
275        
276    heatmap_norm = np.uint8(heatmap_norm)
277    
278    heatmap_resized = cv2.resize(heatmap_norm, (image.shape[1], image.shape[0]))
279    
280    heatmap_color = cv2.applyColorMap(heatmap_resized, cv2.COLORMAP_JET)
281    
282    if image.max() <= 1.0:
283        image = np.uint8(image * 255)
284    
285    image_bgr = cv2.cvtColor(image, cv2.COLOR_RGB2BGR)
286        
287    combined_img = cv2.addWeighted(heatmap_color, 0.7, image_bgr, 0.3, 0)
288    
289    return cv2.cvtColor(combined_img, cv2.COLOR_BGR2RGB)

389def get_predictions(model, dataloader, device=torch.device("cuda" if torch.cuda.is_available() else "cpu")):
390    """
391    Obtains all of the model's predictions and ground truth labels of every image in a given dataset.
392
393    Args:
394        model (nn.Module): The Swin-Xception model making predictions.
395        dataloader (DataLoader): The batched dataset the model is running predictions on.
396        device (torch.device): The device the model is utilising.
397
398    Returns:
399        all_preds (np.array) : Numpy array of all the model's predicted labels.
400        all_labels (np.array) : Numpy array of all of the ground truth labels.
401    """
402
403    model.eval()
404    all_preds = []
405    all_labels = []
406    
407    with torch.no_grad():
408        for images, labels in tqdm(dataloader, desc="Getting predictions"):
409            images = images.to(device)

411            _, preds = outputs.max(1)
412            
413            all_preds.extend(preds.cpu().numpy())
414            all_labels.extend(labels.numpy())
415    
416    return np.array(all_preds), np.array(all_labels)

418def plot_confusion_matrix(y_true, y_pred, class_names, title, save_path=None):
419    """
420    Plot and optionally save confusion matrices.
421
422    Plots a confusion matrix counting actual predictions, and a confusion matrix normalised for percentages.
423
424    Args:
425        y_true (List): List of ground truth labels.
426        y_pred (List): List of model-predicted labels.
427        class_names (List[string]): List of class names.

442    ax1.set_title(f'{title} - Raw Counts')
443    ax1.set_ylabel('True Label')
444    ax1.set_xlabel('Predicted Label')
445    
446    # Percentages
447    sns.heatmap(cm_normalized, annot=True, fmt='.1f', cmap='Blues',
448                xticklabels=class_names, yticklabels=class_names, ax=ax2)
449    ax2.set_title(f'{title} - Percentages (%)')
450    ax2.set_ylabel('True Label')
451    ax2.set_xlabel('Predicted Label')
452    
453    plt.tight_layout()
454    
455    if save_path:
456        plt.savefig(save_path, dpi=300, bbox_inches='tight')

461def print_detailed_metrics(y_true, y_pred, class_names, dataset_name):

463    Outputs:

493def calculate_uar_war(y_true, y_pred, class_names):
494    """
495    Calculate UAR and WAR metrics
496    
497    UAR (Unweighted Average Recall): Mean of per-class recalls (treats all classes equally)
498
499    WAR (Weighted Average Recall): Weighted mean of per-class recalls (weighted by class frequency)
500
501    Args:
502        y_true (List): List of ground truth labels.
503        y_pred (List): List of model-predicted labels.
504        class_names (List[string]): List of class names.
505
506    Returns:
507        uar (float): Unweighted Average Recall
508        war (float): Weighted Average Recall
509        per_class_recall (float)
510    """
511    cm = confusion_matrix(y_true, y_pred)
512    
513    # Per-class recall (same as per-class accuracy for single-label classification)
514    per_class_recall = cm.diagonal() / cm.sum(axis=1)
515    
516    # UAR: Simple average of all class recalls
517    uar = np.mean(per_class_recall) * 100
518    
519    # WAR: Weighted average by class support
520    class_support = cm.sum(axis=1)
521    war = np.average(per_class_recall, weights=class_support) * 100
522    
523    return uar, war, per_class_recall

563def visualise_tsne(
564    model: torch.nn.Module,
565    dataloader: DataLoader,
566    class_names: Optional[list[str]] = None,
567    n_pca_components: int = 50,
568    tsne_perplexity: float = 30.0,
569    tsne_max_iter: int = 1000,
570    device: Optional[str] = None,
571    save_path: Optional[str] = None,
572    title: str = "t-SNE of SwinXception GAP Features",
573) -> tuple[np.ndarray, np.ndarray]:
574    """
575    Extract post-GAP features from SwinXception and visualise with t-SNE.
576
577    The extraction hook attaches after self.avgpool1d so features are
578    768-d vectors — semantically rich, spatially summarised, and compact
579    enough for reliable t-SNE embedding.
580
581    Pipeline:  raw tokens (B, 49, 768)
582                → LayerNorm → AdaptiveAvgPool1d → (B, 768)
583                → PCA (50-d)  → t-SNE (2-d)  → scatter plot
584
585    Args:
586        model (nn.Module): SwinXception instance (eval mode recommended).
587        dataloader (DataLoader): Yields (images, labels) batches.
588        class_names (List[string]): Optional list mapping label indices to strings.
589        n_pca_components (int): PCA dimensionality before t-SNE (set 0 to skip).
590        tsne_perplexity (float): t-SNE perplexity; typically 5–50.
591        tsne_max_iter (int): t-SNE optimisation iterations.
592        device (torch.device): 'cuda', 'cpu', or auto-detected if None.
593        save_path (string): If given, saves the figure to this path.
594        title (string): Plot title.
595
596    Returns:
597        embeddings (List): (N, 2) t-SNE coordinates.
598        labels (List[int]): (N,) ground-truth class indices.
599    """
600    if device is None:
601        device = "cuda" if torch.cuda.is_available() else "cpu"
602
603    model = model.to(device)
604    model.eval()
605
606    # Register a forward hook on avgpool1d to capture GAP output      
607    gap_features: list[torch.Tensor] = []
608
609    def _hook(module, input, output):
610        # output shape: (B, 768, 1)  →  squeeze to (B, 768)
611        gap_features.append(output.squeeze(-1).detach().cpu())
612
613    hook_handle = model.avgpool1d.register_forward_hook(_hook)
614
615    # Forward pass — no gradients needed                              
616    all_labels: list[torch.Tensor] = []

494def calculate_uar_war(y_true, y_pred, class_names):
495    """
496    Calculate UAR and WAR metrics
497    
498    UAR (Unweighted Average Recall): Mean of per-class recalls (treats all classes equally)
499
500    WAR (Weighted Average Recall): Weighted mean of per-class recalls (weighted by class frequency)
501
502    Args:
503        y_true (List): List of ground truth labels.
504        y_pred (List): List of model-predicted labels.
505        class_names (List[string]): List of class names.
506
507    Returns:
508        uar (float): Unweighted Average Recall
509        war (float): Weighted Average Recall
510        per_class_recall (float)
511    """
512    cm = confusion_matrix(y_true, y_pred)
513    
514    # Per-class recall (same as per-class accuracy for single-label classification)
515    per_class_recall = cm.diagonal() / cm.sum(axis=1)
516    
517    # UAR: Simple average of all class recalls
518    uar = np.mean(per_class_recall) * 100
519    
520    # WAR: Weighted average by class support
521    class_support = cm.sum(axis=1)
522    war = np.average(per_class_recall, weights=class_support) * 100
523    
524    return uar, war, per_class_recall

564def visualise_tsne(
565    model: torch.nn.Module,
566    dataloader: DataLoader,
567    class_names: Optional[list[str]] = None,
568    n_pca_components: int = 50,
569    tsne_perplexity: float = 30.0,
570    tsne_max_iter: int = 1000,
571    device: Optional[str] = None,
572    save_path: Optional[str] = None,
573    title: str = "t-SNE of SwinXception GAP Features",
574) -> tuple[np.ndarray, np.ndarray]:
575    """
576    Extract post-GAP features from SwinXception and visualise with t-SNE.
577
578    The extraction hook attaches after self.avgpool1d so features are
579    768-d vectors — semantically rich, spatially summarised, and compact
580    enough for reliable t-SNE embedding.
581
582    Pipeline:  raw tokens (B, 49, 768)
583                → LayerNorm → AdaptiveAvgPool1d → (B, 768)
584                → PCA (50-d)  → t-SNE (2-d)  → scatter plot
585
586    Args:
587        model (nn.Module): SwinXception instance (eval mode recommended).
588        dataloader (DataLoader): Yields (images, labels) batches.
589        class_names (List[string]): Optional list mapping label indices to strings.
590        n_pca_components (int): PCA dimensionality before t-SNE (set 0 to skip).
591        tsne_perplexity (float): t-SNE perplexity; typically 5–50.
592        tsne_max_iter (int): t-SNE optimisation iterations.
593        device (torch.device): 'cuda', 'cpu', or auto-detected if None.
594        save_path (string): If given, saves the figure to this path.
595        title (string): Plot title.
596
597    Returns:
598        embeddings (List): (N, 2) t-SNE coordinates.
599        labels (List[int]): (N,) ground-truth class indices.
600    """
601    if device is None:
602        device = "cuda" if torch.cuda.is_available() else "cpu"
603
604    model = model.to(device)
605    model.eval()
606
607    # Register a forward hook on avgpool1d to capture GAP output      
608    gap_features: list[torch.Tensor] = []
609
610    def _hook(module, input, output):
611        # output shape: (B, 768, 1)  →  squeeze to (B, 768)
612        gap_features.append(output.squeeze(-1).detach().cpu())
613
614    hook_handle = model.avgpool1d.register_forward_hook(_hook)
615
616    # Forward pass — no gradients needed                              
617    all_labels: list[torch.Tensor] = []
618
619    with torch.no_grad():
620        for images, labels in dataloader:
621            images = images.to(device)
622            _ = model(images)          # triggers the hook
623            all_labels.append(labels.cpu())
624
625    hook_handle.remove()
626
627    features = torch.cat(gap_features, dim=0).numpy()   # (N, 768)
628    labels   = torch.cat(all_labels,   dim=0).numpy()   # (N,)
629
630    print(f"Extracted features: {features.shape}  |  samples: {len(labels)}")
631
632    # ------------------------------------------------------------------ #
633    # 3.  Optional PCA pre-reduction                                      #
634    # ------------------------------------------------------------------ #
635    if n_pca_components > 0 and features.shape[1] > n_pca_components:
636        print(f"PCA: {features.shape[1]}-d → {n_pca_components}-d …")
637        pca = PCA(n_components=n_pca_components, random_state=42)
638        features = pca.fit_transform(features)
639        explained = pca.explained_variance_ratio_.sum()
640        print(f"  Explained variance retained: {explained:.1%}")
641
642    # ------------------------------------------------------------------ #
643    # 4.  t-SNE embedding                                                 #
644    # ------------------------------------------------------------------ #
645    print(f"t-SNE: {features.shape[1]}-d → 2-d  "
646          f"(perplexity={tsne_perplexity}, n_iter={tsne_max_iter}) …")
647
648    tsne = TSNE(
649        n_components=2,
650        perplexity=tsne_perplexity,
651        max_iter=tsne_max_iter,
652        random_state=42,
653        init="pca",          # more stable than random init
654        learning_rate="auto",
655    )
656    embeddings = tsne.fit_transform(features)   # (N, 2)
657    print("Done.")
658
659    # ------------------------------------------------------------------ #
660    # 5.  Plot                                                            #
661    # ------------------------------------------------------------------ #
662    n_classes = len(np.unique(labels))
663    cmap      = plt.get_cmap("tab10" if n_classes <= 10 else "tab20")
664    colours   = [cmap(i / max(n_classes - 1, 1)) for i in range(n_classes)]
665
666    fig, ax = plt.subplots(figsize=(10, 8))
667    ax.set_facecolor("#0f0f0f")
668    fig.patch.set_facecolor("#0f0f0f")
669
670    for cls_idx in range(n_classes):
671        mask = labels == cls_idx
672        ax.scatter(
673            embeddings[mask, 0],
674            embeddings[mask, 1],
675            c=[colours[cls_idx]],
676            s=12,
677            alpha=0.75,
678            linewidths=0,
679            label=class_names[cls_idx] if class_names else str(cls_idx),
680        )
681
682    legend = ax.legend(
683        loc="best",
684        framealpha=0.2,
685        edgecolor="white",
686        labelcolor="white",
687        fontsize=9,
688    )
689    ax.set_title(title, color="white", fontsize=13, pad=12)
690    ax.tick_params(colors="grey")
691    for spine in ax.spines.values():
692        spine.set_edgecolor("#333333")
693
694    plt.tight_layout()
695
696    if save_path:
697        plt.savefig(save_path, dpi=150, bbox_inches="tight")
698        print(f"Saved to {save_path}")
699
700    plt.show()
701    return embeddings, labels
