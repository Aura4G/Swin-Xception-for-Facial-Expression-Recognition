  6class PatchEmbedding(nn.Module):
  7    """
  8    The Patch Embedding layer.
  9    """
 10
 11    def __init__(self,
 12                 in_channels:int=3,
 13                 dim:int=96,
 14                 patch_size:int=4):
 15        
 16        """
 17        Initialises a Patch Embedding module.
 18
 19        Args:
 20            in_channels (int): The colour depth of the input image expected. Default: 3
 21            dim (int): The quantity of embedding dimensions to produced. Default: 96
 22            patch_size (int): The height and width of each patch. Default: 4.
 23        """
 24
 25        super(PatchEmbedding, self).__init__()
 26
 27        self.patch_size = patch_size
 28        self.dim = dim
 29
 30        # 2D Convolution produces spatial grid of patches
 31        self.patcher = nn.Conv2d(in_channels=in_channels, out_channels=dim, kernel_size=patch_size, stride=patch_size, padding=0)
 32
 33        # Flatten layer to produce sequence of patches.
 34        self.flatten = nn.Flatten(start_dim=2, end_dim=3)
 35
 36        # Apply normalisation to patches
 37        self.norm = nn.LayerNorm(dim)
 38
 39    def forward(self, x):
 40        """
 41        Takes each image input and creates non-overlapping patches as embeddings before flattening the patches to a sequential input.
 42
 43        Args:
 44            x (torch.Tensor): Input image of shape (B, C, H, W), where H and W 
 45                must be divisible by patch_size.
 46        Returns:
 47            torch.Tensor: Patch sequence of shape (B, N, dim), where
 48                N = (H/patch_size) * (W/patch_size). 
 49
 50        Raises:
 51            AssertionError: If image resolution is not divisible by patch size.
 52        """
 53
 54        image_resolution = x.shape[-1]
 55
 56        assert image_resolution % self.patch_size == 0, \
 57            f"Input image size must be divisible by patch size, \
 58            image shape: {image_resolution}, patch size: {self.patch_size}"
 59        
 60        x_patched = self.patcher(x)
 61
 62        x_flattened = self.flatten(x_patched)
 63
 64        x = x_flattened.permute(0,2,1)
 65
 66        x = self.norm(x)
 67
 68        return x

 71class PatchMerging(nn.Module):
 72    """
 73    The Patch Merging layer proceeding each Swin-Xception layer.
 74    """
 75
 76    def __init__(self, dim):
 77        """
 78        Initialises a Patch Merging module.
 79
 80        Args:
 81        - dim (int): The current quantity of embedding dimensions.
 82        """
 83
 84        super(PatchMerging, self).__init__()
 85
 86        self.dim = dim
 87
 88        # Apply normalisation to concatenated patches
 89        self.norm = nn.LayerNorm(4*dim)
 90
 91        # Linear projection layer to reduce dimensionality before returning double channels
 92        self.reduction = nn.Linear(4*dim, 2*dim, bias=False)
 93
 94    def forward(self, x):
 95        """
 96        Takes sequential tokens from the SwinX block,
 97        rearranges the tokens to achieve the spatial information in the form of a grid,
 98        splits the grid into a 2*2 neighbourhood of patches,
 99        and concatenates the grids, increasing dimensionality by a factor of 4.
100        Linear projection halves spatial grid and doubles channels.
101
102        Args:
103            x (torch.Tensor): Patch sequence of shape (B, N, C), where N = H*W 
104                and H, W are assumed even.    
105    
106        Returns:
107            torch.Tensor: Downsampled Sequence of shape (B, (H/2)*(W/2), 2C).
108
109        Raises:
110            AssertionError: If either the H or W derived from N are not even.
111        """
112
113        # Cast the data structure of the sequence input into variables
114        # B - Batch number
115        # N - Sequence length
116        # C - Channel quantity
117        B, N, C = x.shape
118
119        # Derive the height (H) and width (W) from the square root of N
120        H = W = int(np.sqrt(N))
121
122        assert H % 2 == 0 and W % 2 == 0, f"H and W must be even, got H={H}, W={W}"
123
124        # Reshape sequential tokens to spatial grid
125        x = x.view(B, H, W, C)
126
127        # Split spatial grid of tokens into 4 groups
128        x0 = x[:, 0::2, 0::2, :] # top left
129        x1 = x[:, 1::2, 0::2, :] # top right
130        x2 = x[:, 0::2, 1::2, :] # bottom left
131        x3 = x[:, 1::2, 1::2, :] # bottom right
132
133        # Concatenate the groups of tokens
134        x = torch.cat([x0,x1,x2,x3], dim=-1)
135
136        # Rearrange the spatial grid of patches back into a sequence
137        # flattens spatial grid back to sequence; each token now carries 4C channels from its 2*2 neighbourhood
138        x = x.view(B, -1, 4*C)
139
140        x = self.norm(x)
141
142        # The linear projection squeezes the dimensionality from 4C to 2C
143        x = self.reduction(x)
144
145        return x

148class DepthwiseSeparableConv(nn.Module):
149    """
150    The Depthwise Separable Convolution operator, present in each Depthwise Separable Feed-Forward Network.
151    """
152
153    def __init__(self,
154                 in_channels,
155                 out_channels,
156                 kernel_size:int=3):
157        """
158        Initialises a Depthwise-Separable Convolution layer.
159
160        Args:
161        - in_channels (int): The incoming channel quantity.
162        - out_channels (int): The resulting channel quantity.
163        - kernel_size (int): Kernel of odd size to enable correct symmetric padding. Default: 3.
164        """
165
166        super(DepthwiseSeparableConv, self).__init__()
167
168        # Achieves a depthwise separable convolution, by setting out_channels=in_channels to keep the same number of channels
169        # after the convolution (produces in_channels feature maps). groups=in_channels ensures the kernel is applied to each
170        # channel separately.
171        self.depthwise = nn.Conv2d(in_channels, in_channels, kernel_size, padding=kernel_size//2, groups=in_channels)
172        
173        # Depthwise-separable and pointwise operations are separated by GELU activation
174        self.gelu = nn.GELU()
175
176        # Achieves a pointwise convolution by utilising a kernel size of 1 on the channels.
177        # Channels are not grouped separately
178        self.pointwise = nn.Conv2d(in_channels, out_channels, kernel_size=1)
179
180    def forward(self, x):
181        """
182        Comprises one pair of
183        depthwise-separable and pointwise convolutions; this process expands to/reduces from the MLP
184        ratio in the FFN
185
186        Args:
187            x (torch.Tensor): Spatial grid of patches with dimensions (B, (C/6C), H, W)
188
189        Returns:
190            torch.Tensor: Spatial grid of patches with dimensions (B, (6C/C), H, W)
191        """
192
193        x = self.depthwise(x)
194
195        x = self.gelu(x)
196
197        x = self.pointwise(x)
198
199        return x

202class DepthwiseSeparableFFN(nn.Module):
203    """
204    The Depthwise Separable Feed-Forward Network (The Xception component)
205
206    Replaces the MLP head at the end of each Swin Block.
207    """
208
209    def __init__(self,
210                 dim,
211                 mlp_ratio:int=6,
212                 dropout:float=0.25):
213        """
214        Initialises a DS-FFN
215
216        Args:
217        - dim (int): The embedding dimension quantity
218        - mlp_ratio (int): The expansion factor for which the pointwise convolution widens/narrows the channel count. Default: 6.
219        - dropout (float): The dropout probability inbetween each depthwise separable convolution layer. Default: 0.25.
220        """
221
222        super(DepthwiseSeparableFFN, self).__init__()
223
224        # hidden_dim represents the total number of channels anticipated after the pointwise convolution produces more channels
225        # The number of embeddings in the hidden layer is calculated by multiplying the current number of dimensions by the expansion ratio
226        hidden_dim = int(dim * mlp_ratio)
227
228        # Calls Depthwise Separable Convolution layer, which contains one pair of depthwise-separable and pointwise convolutions
229        # The first layer expands the dimensions by mlp_ratio
230        self.depthwise1 = DepthwiseSeparableConv(dim, hidden_dim, kernel_size=3)
231
232        # Apply dropout
233        self.dropout1 = nn.Dropout(dropout)
234
235        # The second layer reduces the dimensions by the same ratio to achieve the same number of dimensions as before.
236        self.depthwise2 = DepthwiseSeparableConv(hidden_dim, dim, kernel_size=3)
237
238        self.dropout2 = nn.Dropout(dropout)
239
240    def forward(self, x):
241        """
242        Rearranges the sequence of patches into a spatial grid, before expanding and reducing
243        the channel quantity through a pair of depthwise-separable convolution layers.
244
245        Args:
246            x (torch.Tensor): Patch sequence of shape (B, N, C), where N = H*W 
247                and H, W are assumed even. 
248
249        Returns:
250            torch.Tensor: Patch sequence of shape (B, N, C).
251        """
252
253        # Cast the data structure of the sequence input into variables
254        # B - Batch number
255        # N - Sequence length
256        # C - Channel quantity
257        B, N, C = x.shape
258
259        # Derive the height (H) and width (W) from the square root of N
260        H = W = int(np.sqrt(N))
261
262        # Swap the positions of C and N in the input
263        # Replace N with H and W to achieve a spatial grid
264        x = x.transpose(1, 2).reshape(B, C, H, W)
265
266        x = self.depthwise1(x)
267
268        x = self.dropout1(x)
269
270        x = self.depthwise2(x)
271
272        x = self.dropout2(x)
273
274        # Revert changes to the structure of the input
275        x = x.reshape(B, C, N).transpose(1, 2)
276
277        return x

279class SwinXceptionBlock(nn.Module):
280    """
281    Swin-Xception Block.
282
283    Combines the Shifted Window Multi-head Attention Mechanism of Swin
284    with a depthwise-separable feed-forward network (The Xception) in 
285    place of the standard FFN.
286    """
287    def __init__(self,
288                 embedding_dim,
289                 num_heads,
290                 input_resolution,
291                 window_size:int=7,
292                 shift_size:int=0,
293                 mlp_ratio:int=6):
294        """
295        Initialises a Swin-Xception Block.
296
297        Args:
298            embedding_dim (int): Number of input embedding dimensions (C).
299            num_heads (int): Number of attention heads in the Window Multi-Head Self-Attention module.
300            input_resolution (tuple[int, int]): Spatial resolution (H, W) of the input feature map, used to compute attention masks for cyclic shifting.
301            window_size (int): Height and widht of each local attention window. Default: 7.
302            shift_size (int): Number of tokens to cyclically shift the feature map before windowed attention. 0 for W-MSA, window_size//2 for SW-MSA. Default: 0
303            mlp_ratio (int): The expansion factor for the hidden dimension in the Depthwise Separable FFN. Default: 6.
304        """
305
306        super(SwinXceptionBlock, self).__init__()
307
308        self.block = SwinTransformerBlock( # Use timm's Swin Transformer Block
309            dim=embedding_dim,
310            input_resolution=input_resolution,
311            num_heads=num_heads,
312            window_size=window_size,
313            shift_size=shift_size,
314            mlp_ratio=mlp_ratio,
315            proj_drop=0.1,
316            drop_path=0.1
317        )
318
319        self.input_resolution = input_resolution
320
321        self.block.mlp = DepthwiseSeparableFFN(embedding_dim, mlp_ratio) # But replace the mlp head with my DS-FFN
322
323    def forward(self, x): # X input is sequential transformer data
324        """
325        Computes Multi-head Self Attention and Depthwise Separable Convolutions via
326        fixed-size non-overlapping windows containing patches. The feature map is then
327        cyclically shifted to compute attention across window boundaries.
328
329        Args:
330            x (torch.Tensor): Patch sequence of shape (B, N, C), where N = H*W 
331                and H, W are assumed even. 
332
333        Returns:
334            torch.Tensor: Patch sequence of shape (B, N, C), unchanged in shape 
335                from the input after residual addition.
336        """
337
338        # Cast the data structure of the sequence input into variables
339        # B - Batch number
340        # N - Sequence length
341        # C - Channel quantity
342        B, N, C = x.shape
343
344        H, W = self.input_resolution # height and width
345
346        # Reshape sequential tokens to spatial grid
347        x = x.view(B, H, W, C)
348
349        # Swin blocks accept inputs as spatial grid
350        x = self.block(x)
351
352        # back to sequential data
353        x = x.view(B, N, C)
354        
355        return x

357class SwinXception(nn.Module):
358    """
359    The entire backbone for the Swin-Xception architecture. My implementation utilises 6 blocks
360    on the third layer, 3 heads for MSA computation, and an initial embedding count of 96
361    """
362
363    def __init__(self, num_classes:int=7, dropout:float=0.5):
364        """
365        Initialises the Swin-Xception backbone
366
367        Args:
368            num_classes (int): The total classes the model is predicting. Default: 7 for my datasets.
369            dropout (float): The dropout probability before linear projection in the model's head.
370        """
371
372        super(SwinXception, self).__init__()
373
374        # Patch Embedding
375        self.patch_embed = PatchEmbedding(in_channels=3, dim=96, patch_size=4)
376
377        # Stage 1
378        self.layer1 = nn.ModuleList([SwinXceptionBlock(96, num_heads=3, input_resolution=(56, 56), shift_size=0 if i % 2 == 0 else 3) for i in range(2)]) # Parity to enable window shift
379        self.merge1 = PatchMerging(96) # Reduce dimensions, increase channels
380
381        # Stage 2
382        self.layer2 = nn.ModuleList([SwinXceptionBlock(192, num_heads=6, input_resolution=(28, 28), shift_size=0 if i % 2 == 0 else 3) for i in range(2)])
383        self.merge2 = PatchMerging(192)
384
385        # Stage 3 - Utilises 6 blocks
386        self.layer3 = nn.ModuleList([SwinXceptionBlock(384, num_heads=12, input_resolution=(14, 14), shift_size=0 if i % 2 == 0 else 3) for i in range(6)])
387        self.merge3 = PatchMerging(384)
388
389        # Stage 4
390        self.layer4 = nn.ModuleList([SwinXceptionBlock(768, num_heads=24, input_resolution=(7, 7), shift_size=0 if i % 2 == 0 else 3) for i in range(2)])
391
392        # Normalisation
393        self.norm = nn.LayerNorm(768)
394        
395        # Average Pooling
396        self.avgpool1d = nn.AdaptiveAvgPool1d(output_size=1)
397
398        # MLP Head
399        self.head = nn.Sequential(
400            nn.Dropout(dropout), # Apply dropout before Linear Projection to class values
401            nn.Linear(768, num_classes)
402        )
403
404    def forward(self, x):
405        """
406        Args:
407            x (torch.Tensor): Input image of shape (B, C, H, W).
408
409        Returns:
410            torch.Tensor: Class logits of shape (B, num_classes), where each value 
411                is the unnormalised score for the corresponding class.
412        """
413
414        # x = [B, 3, 224, 224]
415        x = self.patch_embed(x) # x = [B, 3136, 96]
416
417        # Stage 1: 2 Swin-X blocks
418        for block in self.layer1:
419            x = block(x)
420        x = self.merge1(x) # x = [B, 784, 192]
421
422        # Stage 2: 2 Swin-X blocks
423        for block in self.layer2:
424            x = block(x)
425        x = self.merge2(x) # x = [B, 196, 384]
426
427        # Stage 3: 6 Swin-X blocks
428        for block in self.layer3:
429            x = block(x)
430        x = self.merge3(x) # x = [B, 49, 768]
431
432        # Stage 4: 2 Swin-X blocks
433        for block in self.layer4:
434            x = block(x)
435
436        # Normalisation and Average Pooling
437        x = self.norm(x)
438        x = x.transpose(1, 2) # x = [B, 768, 49]
439        x = self.avgpool1d(x) # x = [B, 768, 1]
440        x = torch.flatten(x, 1) # x = [B, 768]
441
442        # Linear Projection
443        x = self.head(x) # x = [B, 7]
444
445        return x

204    """
205    The Depthwise Separable Feed-Forward Network (The Xception component)
206
207    Replaces the MLP head at the end of each Swin Block.
208    """
209
210    def __init__(self,
211                 dim,
212                 mlp_ratio:int=6,
213                 dropout:float=0.25):
214        """
215        Initialises a DS-FFN
216
217        Args:
218        - dim (int): The embedding dimension quantity
219        - mlp_ratio (int): The expansion factor for which the pointwise convolution widens/narrows the channel count. Default: 6.
220        - dropout (float): The dropout probability inbetween each depthwise separable convolution layer. Default: 0.25.

222

224
225        # hidden_dim represents the total number of channels anticipated after the pointwise convolution produces more channels
226        # The number of embeddings in the hidden layer is calculated by multiplying the current number of dimensions by the expansion ratio
227        hidden_dim = int(dim * mlp_ratio)
228

235

244        the channel quantity through a pair of depthwise-separable convolution layers.

248                and H, W are assumed even.

252        """

359    """
360    The entire backbone for the Swin-Xception architecture. My implementation utilises 6 blocks
361    on the third layer, 3 heads for MSA computation, and an initial embedding count of 96
362    """
363
364    def __init__(self, num_classes:int=7, dropout:float=0.5):
365        """
366        Initialises the Swin-Xception backbone
367

369            num_classes (int): The total classes the model is predicting. Default: 7 for my datasets.
370            dropout (float): The dropout probability before linear projection in the model's head.
371        """
372
373        super(SwinXception, self).__init__()
374
375        # Patch Embedding
376        self.patch_embed = PatchEmbedding(in_channels=3, dim=96, patch_size=4)
377
378        # Stage 1
379        self.layer1 = nn.ModuleList([SwinXceptionBlock(96, num_heads=3, input_resolution=(56, 56), shift_size=0 if i % 2 == 0 else 3) for i in range(2)]) # Parity to enable window shift
380        self.merge1 = PatchMerging(96) # Reduce dimensions, increase channels
381
382        # Stage 2
383        self.layer2 = nn.ModuleList([SwinXceptionBlock(192, num_heads=6, input_resolution=(28, 28), shift_size=0 if i % 2 == 0 else 3) for i in range(2)])
384        self.merge2 = PatchMerging(192)
385

387        self.layer3 = nn.ModuleList([SwinXceptionBlock(384, num_heads=12, input_resolution=(14, 14), shift_size=0 if i % 2 == 0 else 3) for i in range(6)])
388        self.merge3 = PatchMerging(384)
389
390        # Stage 4
391        self.layer4 = nn.ModuleList([SwinXceptionBlock(768, num_heads=24, input_resolution=(7, 7), shift_size=0 if i % 2 == 0 else 3) for i in range(2)])
392
393        # Normalisation
394        self.norm = nn.LayerNorm(768)
395        
396        # Average Pooling
397        self.avgpool1d = nn.AdaptiveAvgPool1d(output_size=1)
398
399        # MLP Head
400        self.head = nn.Sequential(
401            nn.Dropout(dropout), # Apply dropout before Linear Projection to class values
402            nn.Linear(768, num_classes)
403        )

405    def forward(self, x):
406        """
407        Args:
408            x (torch.Tensor): Input image of shape (B, C, H, W).
409
410        Returns:
411            torch.Tensor: Class logits of shape (B, num_classes), where each value 
412                is the unnormalised score for the corresponding class.
413        """
414
415        # x = [B, 3, 224, 224]
416        x = self.patch_embed(x) # x = [B, 3136, 96]
417
418        # Stage 1: 2 Swin-X blocks
419        for block in self.layer1:
420            x = block(x)
421        x = self.merge1(x) # x = [B, 784, 192]
422
423        # Stage 2: 2 Swin-X blocks
424        for block in self.layer2:
425            x = block(x)
426        x = self.merge2(x) # x = [B, 196, 384]
427
428        # Stage 3: 6 Swin-X blocks
429        for block in self.layer3:
430            x = block(x)
431        x = self.merge3(x) # x = [B, 49, 768]
432
433        # Stage 4: 2 Swin-X blocks
434        for block in self.layer4:
435            x = block(x)
436
437        # Normalisation and Average Pooling
438        x = self.norm(x)
439        x = x.transpose(1, 2) # x = [B, 768, 49]
440        x = self.avgpool1d(x) # x = [B, 768, 1]
441        x = torch.flatten(x, 1) # x = [B, 768]
